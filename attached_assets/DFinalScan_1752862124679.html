function formatAge(timestamp) {

  const now = Date.now();

  const ageMs = now - timestamp;

  const seconds = Math.floor(ageMs / 1000);

  const minutes = Math.floor(seconds / 60);

  const hours = Math.floor(minutes / 60);

  const days = Math.floor(hours / 24);

  if (days > 0) {

    return `${days}d ${hours % 24}h`;

  } else if (hours > 0) {

    return `${hours}h ${minutes % 60}m`;

  } else if (minutes > 0) {

    return `${minutes}m`;

  } else {

    return `${seconds}s`;

  }

}

function calculateConfidence(pair) {

  const now = Date.now();

  const liquidity = pair.liquidity?.usd ?? 0;

  const priceChange = pair.priceChange?.h24 ?? 0;

  const volume = pair.volume?.h24 ?? 0;

  const ageHours = (now - pair.pairCreatedAt) / (1000 * 60 * 60);

  // Calculate factors with bounds

  const priceMomentum = Math.min(100, priceChange * 1.5);

  const volumeRatio = liquidity > 0 ? Math.min(100, (volume / liquidity) * 500) : 0;

  const liquidityScore = Math.max(0, 100 - (liquidity / 1000));

  const tokenAge = Math.max(0, 100 - ageHours);

  // Weights

  const weights = {

    priceMomentum: 0.3,

    volumeRatio: 0.3,

    liquidityScore: 0.2,

    tokenAge: 0.2

  };

  let confidence = 0;

  confidence += priceMomentum * weights.priceMomentum;

  confidence += volumeRatio * weights.volumeRatio;

  confidence += liquidityScore * weights.liquidityScore;

  confidence += tokenAge * weights.tokenAge;

  // Ensure confidence is between 0 and 100

  confidence = Math.max(0, Math.min(100, confidence));

  return confidence.toFixed(1); // Return as string with 1 decimal

}

function calculateMarketCap(close_price, total_supply) {

  if (!close_price || !total_supply) return 0;

  return close_price * total_supply;

}

function calculateRSI(close_prices, window = 14) {

  if (!close_prices || close_prices.length < window + 1) return "N/A";

 

  const deltas = [];

  for (let i = 1; i < close_prices.length; i++) {

    deltas.push(close_prices[i] - close_prices[i - 1]);

  }

 

  // Separate gains and losses

  const gains = [];

  const losses = [];

  for (let i = 0; i < window; i++) {

    if (deltas[i] > 0) gains.push(deltas[i]);

    if (deltas[i] < 0) losses.push(Math.abs(deltas[i]));

  }

 

  if (losses.length === 0) return "100.00";

 

  const avgGain = gains.reduce((sum, gain) => sum + gain, 0) / window;

  const avgLoss = losses.reduce((sum, loss) => sum + loss, 0) / window;

 

  const rs = avgGain / avgLoss;

  const rsi = 100 - (100 / (1 + rs));

  return rsi.toFixed(2);

}

function calculateATR(high_prices, low_prices, close_prices, window = 14) {

  if (!high_prices || !low_prices || !close_prices ||

      high_prices.length < window ||

      low_prices.length < window ||

      close_prices.length < window) {

    return "N/A";

  }

 

  const trueRanges = [];

  for (let i = 1; i < window; i++) {

    const high = high_prices[i];

    const low = low_prices[i];

    const prevClose = close_prices[i - 1];

    const tr = Math.max(

      high - low,

      Math.abs(high - prevClose),

      Math.abs(low - prevClose)

    );

    trueRanges.push(tr);

  }

 

  const atr = trueRanges.reduce((sum, tr) => sum + tr, 0) / window;

  return atr.toFixed(2);

}

function calculateSMA(close_prices, window = 50) {

  if (!close_prices || close_prices.length < window) return "N/A";

  const prices = close_prices.slice(-window);

  const sum = prices.reduce((a, b) => a + b, 0);

  return (sum / window).toFixed(2);

}

function calculateEMA(close_prices, span = 50) {

  if (!close_prices || close_prices.length < span) return "N/A";

 

  const k = 2 / (span + 1);

  let ema = close_prices[0];

 

  for (let i = 1; i < close_prices.length; i++) {

    ema = close_prices[i] * k + ema * (1 - k);

  }

 

  return ema.toFixed(2);

}

      const atr = pair.atr;

      const ema_50 = pair.ema_50;

      }

      return `

        <tr class="hover:bg-gray-750">

          <td class="py-3 px-4 whitespace-nowrap">

            <div class="flex items-center">

function handleAutoRefresh(event) {

  if (event.target.checked) {

    autoRefreshInterval = setInterval(runScan, CONFIG.AUTO_REFRESH_INTERVAL);

  }

}

function exportData() {

  if (currentData.length === 0) {

    showNotification('No data to export!');

    return;

  }

  const csvRows = [

    ['Pair', 'Chain', '24h Change (%)', '24h Volume (USD)', 'Liquidity (USD)', 'Age (hours)', 'Confidence', 'Market Cap (USD)', 'RSI (14)', 'ATR (14)', 'SMA 50 (USD)', 'EMA 50 (USD)']

  ];

  currentData.forEach(pair => {

    csvRows.push([

      `${pair.baseToken.symbol}/${pair.quoteToken.symbol}`,

      pair.chainId,

      `${pair.priceChange?.h24 ?? 0}`,

      `${pair.volume?.h24 ?? 0}`,

      `${pair.liquidity?.usd ?? 0}`,

      `${(Date.now() - pair.pairCreatedAt) / 3600000}`,

      pair.confidence,

      pair.market_cap,

      pair.rsi,

      pair.atr,

      pair.sma_50,

      pair.ema_50

    ]);

  });

  const csvContent = csvRows.map(row => row.join(',')).join('\n');

  const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });

  const link = document.createElement("a");

 

  if (navigator.msSaveBlob) { // IE 10+

    navigator.msSaveBlob(blob, "crypto_pump_radar.csv");

  } else {

    const url = URL.createObjectURL(blob);

    link.setAttribute("href", url);

    link.setAttribute("download", "crypto_pump_radar.csv");

    link.style.visibility = 'hidden';

    document.body.appendChild(link);

    link.click();

    document.body.removeChild(link);

  }

 

  showNotification('Data exported to CSV file!');

}

function showNotification(message) {

  notificationText.textContent = message;

  notification.classList.add('show');

 

  setTimeout(() => {

    notification.classList.remove('show');

  }, 3000);

}
